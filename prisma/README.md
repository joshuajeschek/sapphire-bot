## Database Setup
```sql
-- CREATE DATABASE PROJECTNAME;

\connect PROJECTNAME;

CREATE SCHEMA IF NOT EXISTS dev;
CREATE SCHEMA IF NOT EXISTS prod;

-- admin user
GRANT ALL ON DATABASE PROJECTNAME TO jjeschek;
GRANT ALL ON SCHEMA dev, prod TO jjeschek;
GRANT ALL ON ALL TABLES IN SCHEMA dev, prod TO jjeschek;

-- dev user
CREATE USER devPROJECTNAME WITH ENCRYPTED PASSWORD '***';
GRANT USAGE ON SCHEMA dev TO devPROJECTNAME;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA dev TO devPROJECTNAME;

-- prod user
CREATE USER PROJECTNAME WITH ENCRYPTED PASSWORD '***';
GRANT USAGE ON SCHEMA prod TO PROJECTNAME;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA prod TO PROJECTNAME;
```
